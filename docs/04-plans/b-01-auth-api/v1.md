# Auth API 实现计划

> **For agentic workers:** 必需子技能：`superpowers:subagent-driven-development`（推荐）或 `superpowers:executing-plans`。步骤使用复选框（`- [ ]`）语法追踪。

**目标：** 集成 Better Auth，实现邮箱+密码注册/登录/登出/会话查询 API，提供 Hono 认证中间件。

**架构：** 使用 Better Auth 框架 + Drizzle Adapter 对接 PostgreSQL；通过 `IAuthProvider` 接口抽象，实现类 `BetterAuthProvider` 封装 Better Auth API 调用；Hono 路由统一挂载 `/api/auth/**` 到 Better Auth handler；独立速率限制中间件保护认证端点。

**技术栈：** Better Auth, better-auth-drizzle-adapter, Drizzle ORM, Hono, bcrypt (Better Auth 内置)

// packages/server/src/middleware/rate-limit.ts
import { rateLimiter } from 'hono-rate-limiter'
import type { MiddlewareHandler } from 'hono'

function createAuthRateLimit(windowMs: number, max: number): MiddlewareHandler {
  return rateLimiter({
    windowMs,
    max,
    keyGenerator: (c) => c.req.header('x-forwarded-for') ?? c.req.header('x-real-ip') ?? 'unknown',
    handler: (c) => {
      c.header('Retry-After', String(Math.ceil(windowMs / 1000)))
      c.header('X-RateLimit-Limit', String(max))
      c.header('X-RateLimit-Remaining', '0')
      return c.json({ error: 'Too many requests, please try again later' }, 429)
    },
  }) as MiddlewareHandler
}

/** 登录端点限速：5 次/分钟/IP */
export const signInRateLimit = createAuthRateLimit(60 * 1000, 5)

/** 注册端点限速：5 次/分钟/IP */
export const signUpRateLimit = createAuthRateLimit(60 * 1000, 5)

/** 登出端点限速：20 次/分钟/IP */
export const signOutRateLimit = createAuthRateLimit(60 * 1000, 20)

/** 会话查询端点限速：60 次/分钟/IP */
export const sessionRateLimit = createAuthRateLimit(60 * 1000, 60)
```

// packages/server/src/providers/BetterAuthProvider.ts
import { auth } from '../auth.js'
import {
  type IAuthProvider,
  type SignUpCredentials,
  type SignInCredentials,
  type AuthResult,
  type Session,
  ConflictError,
  ValidationError,
  AuthError,
} from '../interfaces/IAuthProvider.js'
import type { MiddlewareHandler } from 'hono'

function toUser(betterUser: Record<string, unknown>) {
  return {
    id: String(betterUser.id),
    email: String(betterUser.email),
    name: betterUser.name ? String(betterUser.name) : null,
    avatar: betterUser.avatar ? String(betterUser.avatar) : null,
    createdAt: new Date(String(betterUser.createdAt)),
  }
}

function toSession(betterSession: Record<string, unknown>): Session {
  return {
    id: String(betterSession.id),
    userId: String(betterSession.userId),
    expiresAt: new Date(String(betterSession.expiresAt)),
  }
}

export class BetterAuthProvider implements IAuthProvider {
  async signUp(credentials: SignUpCredentials): Promise<AuthResult> {
    try {
      const result = await auth.api.signUpEmail({
        body: {
          email: credentials.email,
          password: credentials.password,
          name: credentials.name ?? '',
        },
      })

      if (!result || !result.user || !result.session) {
        throw new AuthError('Registration failed')
      }

      return {
        user: toUser(result.user as Record<string, unknown>),
        session: toSession(result.session as Record<string, unknown>),
      }
    } catch (err: any) {
      if (err?.statusCode === 400 && err?.message?.toLowerCase().includes('password')) {
        throw new ValidationError('Password must be at least 8 characters')
      }
      if (err?.statusCode === 409 || err?.message?.toLowerCase().includes('already exists')) {
        throw new ConflictError('Email already registered')
      }
      if (err instanceof ConflictError || err instanceof ValidationError) {
        throw err
      }
      throw new AuthError('Registration failed')
    }
  }

  async signIn(credentials: SignInCredentials): Promise<AuthResult> {
    try {
      const result = await auth.api.signInEmail({
        body: {
          email: credentials.email,
          password: credentials.password,
        },
      })

      if (!result || !result.user || !result.session) {
        throw new AuthError('Invalid email or password')
      }

      return {
        user: toUser(result.user as Record<string, unknown>),
        session: toSession(result.session as Record<string, unknown>),
      }
    } catch (err: any) {
      if (err instanceof AuthError) throw err
      throw new AuthError('Invalid email or password')
    }
  }

  async signOut(request: Request): Promise<void> {
    try {
      await auth.api.signOut({
        headers: request.headers,
      })
    } catch {
      // 幂等：即使 session 无效也静默成功
    }
  }

  async getSession(request: Request): Promise<Session | null> {
    try {
      const result = await auth.api.getSession({
        headers: request.headers,
      })
      if (!result || !result.session) return null
      return toSession(result.session as Record<string, unknown>)
    } catch {
      return null
    }
  }

  middleware(): MiddlewareHandler {
    return async (c, next) => {
      const session = await this.getSession(c.req.raw)
      if (!session) {
        return c.json({ error: 'Unauthorized' }, 401)
      }
      // 注入 context（后续 handler 通过 c.get('user') / c.get('session') 读取）
      c.set('session', session)
      // user 需要额外查询，这里先注入 session；若需要 user 可在中间件内再查
      await next()
    }
  }
}
```

// packages/server/src/middleware/auth.ts
import { createMiddleware } from 'hono/factory'
import { auth } from '../auth.js'
import type { User, Session } from '../interfaces/IAuthProvider.js'

function toUser(betterUser: Record<string, unknown>): User {
  return {
    id: String(betterUser.id),
    email: String(betterUser.email),
    name: betterUser.name ? String(betterUser.name) : null,
    avatar: betterUser.avatar ? String(betterUser.avatar) : null,
    createdAt: new Date(String(betterUser.createdAt)),
  }
}

function toSession(betterSession: Record<string, unknown>): Session {
  return {
    id: String(betterSession.id),
    userId: String(betterSession.userId),
    expiresAt: new Date(String(betterSession.expiresAt)),
  }
}

/**
 * Hono 认证中间件。
 * 验证 session，有效时将 user / session 注入 Context；无效时返回 401。
 */
export const requireAuth = createMiddleware(async (c, next) => {
  try {
    const result = await auth.api.getSession({
      headers: c.req.raw.headers,
    })

    if (!result || !result.user || !result.session) {
      return c.json({ error: 'Unauthorized' }, 401)
    }

    c.set('user', toUser(result.user as Record<string, unknown>))
    c.set('session', toSession(result.session as Record<string, unknown>))
    await next()
  } catch {
    return c.json({ error: 'Unauthorized' }, 401)
  }
})
```

// packages/server/src/providers/BetterAuthProvider.test.ts
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { BetterAuthProvider } from './BetterAuthProvider.js'
import { auth } from '../auth.js'

const provider = new BetterAuthProvider()

describe('BetterAuthProvider', () => {
  const testEmail = `test-${Date.now()}@example.com`
  const password = 'securePass123'

  it('should sign up a new user', async () => {
    const result = await provider.signUp({ email: testEmail, password, name: 'Alice' })
    expect(result.user.email).toBe(testEmail)
    expect(result.user.name).toBe('Alice')
    expect(result.session).toBeDefined()
    expect(result.session.userId).toBe(result.user.id)
  })

  it('should reject duplicate email with ConflictError', async () => {
    await expect(provider.signUp({ email: testEmail, password })).rejects.toThrow('Email already registered')
  })

  it('should reject weak password with ValidationError', async () => {
    await expect(provider.signUp({ email: `other-${Date.now()}@example.com`, password: '123' })).rejects.toThrow('Password must be at least 8 characters')
  })

  it('should sign in with valid credentials', async () => {
    const result = await provider.signIn({ email: testEmail, password })
    expect(result.user.email).toBe(testEmail)
    expect(result.session).toBeDefined()
  })

  it('should reject invalid credentials with AuthError', async () => {
    await expect(provider.signIn({ email: testEmail, password: 'wrongpassword' })).rejects.toThrow('Invalid email or password')
  })

  it('should get session after sign in', async () => {
    const signInResult = await provider.signIn({ email: testEmail, password })
    // 构造一个模拟 Request 携带 Cookie
    const cookie = `goferbot.session=${signInResult.session.id}`
    const request = new Request('http://localhost', { headers: { cookie } })
    const session = await provider.getSession(request)
    expect(session).not.toBeNull()
    expect(session!.userId).toBe(signInResult.user.id)
  })

  it('should return null for invalid session', async () => {
    const request = new Request('http://localhost', { headers: { cookie: 'goferbot.session=invalid-token' } })
    const session = await provider.getSession(request)
    expect(session).toBeNull()
  })

  it('should sign out and invalidate session', async () => {
    const signInResult = await provider.signIn({ email: testEmail, password })
    const cookie = `goferbot.session=${signInResult.session.id}`
    const request = new Request('http://localhost', { headers: { cookie } })
    await provider.signOut(request)
    const session = await provider.getSession(request)
    expect(session).toBeNull()
  })
})
```
