- [ ] **步骤 1: 创建 tabs.ts，使用 Tab 类型**

```typescript
// packages/webui/src/stores/tabs.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Tab, TabType } from '@/types'

// 单例类型集合
const SINGLETON_TYPES: Set<TabType> = new Set([
  'knowledgeBase', 'history', 'settings', 'recycleBin'
])

export const useTabsStore = defineStore('tabs', () => {
  const tabs = ref<Tab[]>([
    { id: 'home', type: 'chat', title: '首页', sessionId: undefined, closable: false },
  ])
  const activeTabId = ref<string>('home')

  const activeTab = computed(
    () => tabs.value.find((t) => t.id === activeTabId.value) ?? null
  )

  function addTab(type: TabType, sessionId?: string, title?: string) {
    // 单例检查
    if (SINGLETON_TYPES.has(type)) {
      const existing = tabs.value.find((t) => t.type === type)
      if (existing) {
        activeTabId.value = existing.id
        return existing
      }
    }
    const id = type === 'chat' ? `tab-${Date.now()}` : `tab-${type}`
    const newTab: Tab = {
      id,
      type,
      title: title ?? getDefaultTitle(type),
      sessionId,
      closable: true,
    }
    tabs.value.push(newTab)
    activeTabId.value = id
    return newTab
  }

  function closeTab(tabId: string) {
    const idx = tabs.value.findIndex((t) => t.id === tabId)
    if (idx === -1) return
    const tab = tabs.value[idx]
    if (!tab.closable) return
    tabs.value.splice(idx, 1)
    if (activeTabId.value === tabId) {
      const newIdx = Math.max(0, idx - 1)
      activeTabId.value = tabs.value[newIdx]?.id ?? 'home'
    }
  }

  function switchTab(tabId: string) {
    activeTabId.value = tabId
  }

  function renameTab(tabId: string, title: string) {
    const tab = tabs.value.find((t) => t.id === tabId)
    if (tab) tab.title = title
  }

  function updateHomeTabSession(sessionId: string, title: string) {
    const homeTab = tabs.value.find((t) => t.id === 'home')
    if (homeTab) {
      homeTab.sessionId = sessionId
      homeTab.title = title
    }
  }

  // 更新当前激活标签的 session 关联（用于新会话创建时）
  function updateActiveTabSession(sessionId: string, title: string) {
    const tab = tabs.value.find((t) => t.id === activeTabId.value)
    if (tab) {
      tab.sessionId = sessionId
      tab.title = title
    }
  }

  return {
    tabs, activeTabId, activeTab,
    addTab, closeTab, switchTab, renameTab,
    updateHomeTabSession, updateActiveTabSession,
  }
})

function getDefaultTitle(type: TabType): string {
  const map: Record<TabType, string> = {
    chat: '新会话',
    knowledgeBase: '知识库',
    history: '历史记录',
    settings: '设置',
    recycleBin: '回收站',
  }
  return map[type] ?? '未命名'
}
```
