## 步骤 2：实现 StorageProvider Factory Provider

**动作**：新建 `packages/server/src/processors/storage/storage.provider.ts`，负责从 `ConfigService` 读取 MinIO 配置、实例化 `MinIOStorageProvider`、调用 `initialize()`。

```typescript
import { Provider } from '@nestjs/common'
import { ConfigService } from '@nestjs/config'
import { MinIOStorageProvider } from '../../storage/minio.js'
import type { MinIOConfig } from '../../config/storage.js'

export const STORAGE_PROVIDER = 'STORAGE_PROVIDER'

function parseEndpoint(endpoint: string): { host: string; port?: number; useSSL: boolean } {
  const useSSL = endpoint.startsWith('https://')
  const clean = endpoint.replace(/^https?:\/\//, '')
  const portMatch = clean.match(/:(\d+)$/)
  const host = clean.replace(/:\d+$/, '')
  const port = portMatch ? parseInt(portMatch[1], 10) : undefined
  return { host, port, useSSL }
}

export const storageProvider: Provider = {
  provide: STORAGE_PROVIDER,
  useFactory: async (configService: ConfigService): Promise<MinIOStorageProvider> => {
    const endpoint = configService.getOrThrow<string>('MINIO_ENDPOINT')
    const accessKey = configService.getOrThrow<string>('MINIO_ACCESS_KEY')
    const secretKey = configService.getOrThrow<string>('MINIO_SECRET_KEY')
    const bucket = configService.get<string>('MINIO_BUCKET') ?? 'goferbot-files'
    const region = configService.get<string>('MINIO_REGION') ?? 'us-east-1'

    const { host, port, useSSL } = parseEndpoint(endpoint)

    const config: MinIOConfig = {
      endpoint: host,
      port,
      accessKey,
      secretKey,
      bucket,
      region,
      useSSL,
    }

    const provider = new MinIOStorageProvider(config)
    await provider.initialize()
    return provider
  },
  inject: [ConfigService],
}
```

**验证**：文件存在且 `tsc --noEmit` 无类型错误（后续统一执行）。

## 步骤 3：实现 StorageService

**动作**：新建 `packages/server/src/processors/storage/storage.service.ts`，注入 `STORAGE_PROVIDER`，实现 `IStorageProvider` 与 `StorageService` 专属方法。

```typescript
import { Injectable, Inject } from '@nestjs/common'
import { Readable } from 'stream'
import type {
  IStorageProvider,
  StorageMeta,
  StorageUrlOptions,
} from '../../interfaces/IStorageProvider.js'
import { ValidationError, StorageError } from '../../interfaces/errors.js'
import { MinIOStorageProvider } from '../../storage/minio.js'
import { STORAGE_PROVIDER } from './storage.provider.js'

@Injectable()
export class StorageService implements IStorageProvider {
  constructor(
    @Inject(STORAGE_PROVIDER)
    private readonly provider: MinIOStorageProvider
  ) {}

  async uploadFile(buffer: Buffer, key: string, mimeType: string): Promise<StorageMeta> {
    if (!buffer || buffer.length === 0) {
      throw new ValidationError('上传文件 buffer 不能为空')
    }
    const stream = Readable.from(buffer)
    return this.provider.upload(key, stream, mimeType)
  }

  async downloadFile(key: string): Promise<ReadableStream<Uint8Array>> {
    return this.provider.download(key)
  }

  async deleteFile(key: string): Promise<void> {
    return this.provider.delete(key)
  }

  async getUrl(key: string): Promise<string> {
    return this.provider.getUrl(key)
  }

  async getPresignedUploadUrl(key: string, expiry = 3600): Promise<string> {
    return this.provider.getPresignedUploadUrl(key, expiry)
  }

  // IStorageProvider 兼容方法（直接透传）
  async upload(
    key: string,
    stream: ReadableStream<Uint8Array> | NodeJS.ReadableStream,
    contentType: string
  ): Promise<StorageMeta> {
    return this.provider.upload(key, stream, contentType)
  }

  async download(key: string): Promise<ReadableStream<Uint8Array>> {
    return this.provider.download(key)
  }

  async delete(key: string): Promise<void> {
    return this.provider.delete(key)
  }

  async getUrl(key: string, options?: StorageUrlOptions): Promise<string> {
    return this.provider.getUrl(key, options)
  }
}
```

**验证**：文件存在，方法签名与 `api-spec.md` 一致。

## 风险与注意事项

1. **i-08 未完成时的阻塞**：若 NestJS 模块结构尚未建立，步骤 5 无法执行。需等待 i-08 完成后合并。
2. **ConfigModule 未全局注册**：`ConfigModule.forRoot({ isGlobal: true })` 必须在 `AppModule` 中注册，否则 `StorageModule` 导入的 `ConfigModule` 无法解析 `ConfigService`。
3. **MINIO_ENDPOINT 格式**：支持 `http://localhost:9000`、`https://minio.example.com`、`minio.example.com:9000` 三种格式，`parseEndpoint` 已处理。
4. **Buffer → Stream 转换**：`uploadFile` 使用 `Readable.from(buffer)`，依赖 Node.js `stream` 模块，确保运行环境为 Node.js 16+。
