# Session Store 实现计划

> **For agentic workers:** 必需子技能：superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans。步骤使用复选框（`- [ ]`）语法追踪。

**目标：** 扩展 `packages/web/src/stores/chat.ts`（Zustand），添加会话列表管理能力（sessions CRUD），替代旧 Pinia session store。

**架构：** 在现有 ChatState 接口上新增 `sessions` / `isLoadingSessions` / `error` 三个状态字段，以及 4 个同步 actions + 4 个异步 actions。异步 actions 通过 `alovaInstance` 调用 `api/chat.ts` 中的方法，再调用同步 actions 更新本地状态。`sendMessage` 仅保留骨架（f-44 实现）。

**技术栈：** Zustand 5（`create((set, get) => ...)`）、Alova（`.send()` 直接调用）、TypeScript、Vitest

```typescript
import { create } from 'zustand'
import type { Message, Session } from '@goferbot/data'

interface ChatState {
  // ===== 已有字段 =====
  activeSession: Session | null
  messages: Message[]
  isLoadingHistory: boolean
  isStreaming: boolean
  streamingContent: string

  // ===== 新增字段 =====
  sessions: Session[]
  isLoadingSessions: boolean
  error: string | null

  // ===== 已有 actions =====
  setActiveSession: (session: Session | null) => void
  setMessages: (messages: Message[]) => void
  appendMessage: (message: Message) => void
  setIsLoadingHistory: (v: boolean) => void
  setIsStreaming: (v: boolean) => void
  appendStreamContent: (chunk: string) => void
  flushStreamContent: () => void
  clearChat: () => void

  // ===== 新增同步 actions =====
  setSessions: (sessions: Session[]) => void
  addSession: (session: Session) => void
  removeSession: (id: string) => void
  updateSession: (id: string, updates: Partial<Session>) => void
  clearError: () => void

  // ===== 新增异步 actions（骨架，后续任务实现） =====
  loadSessions: () => Promise<void>
  createSession: () => Promise<Session | undefined>
  renameSession: (id: string, title: string) => Promise<void>
  deleteSession: (id: string) => Promise<void>
  sendMessage: (content: string, knowledgeBaseIds?: string[]) => Promise<void>
}

export const useChatStore = create<ChatState>((set) => ({
  // ===== 已有状态 =====
  activeSession: null,
  messages: [],
  isLoadingHistory: false,
  isStreaming: false,
  streamingContent: '',

  // ===== 新增状态 =====
  sessions: [],
  isLoadingSessions: false,
  error: null,

  // ===== 已有 actions（保持不变） =====
  setActiveSession: (session) => set({ activeSession: session }),
  setMessages: (messages) => set({ messages }),
  appendMessage: (message) =>
    set((s) => ({ messages: [...s.messages, message] })),
  setIsLoadingHistory: (v) => set({ isLoadingHistory: v }),
  setIsStreaming: (v) => set({ isStreaming: v }),
  appendStreamContent: (chunk) =>
    set((s) => ({ streamingContent: s.streamingContent + chunk })),
  flushStreamContent: () =>
    set((s) => {
      if (!s.streamingContent) return s
      const assistantMsg: Message = {
        id: `msg-${Date.now()}`,
        sessionId: s.activeSession?.id ?? '',
        role: 'assistant',
        content: s.streamingContent,
        createdAt: new Date().toISOString(),
      }
      return {
        messages: [...s.messages, assistantMsg],
        streamingContent: '',
      }
    }),
  clearChat: () =>
    set({
      activeSession: null,
      messages: [],
      streamingContent: '',
      isStreaming: false,
    }),

  // ===== 新增同步 actions =====
  setSessions: (sessions) => set({ sessions }),
  addSession: (session) => set((s) => ({ sessions: [session, ...s.sessions] })),
  removeSession: (id) =>
    set((s) => ({ sessions: s.sessions.filter((sess) => sess.id !== id) })),
  updateSession: (id, updates) =>
    set((s) => ({
      sessions: s.sessions.map((sess) =>
        sess.id === id ? { ...sess, ...updates } : sess,
      ),
    })),
  clearError: () => set({ error: null }),

  // ===== 新增异步 actions（骨架） =====
  loadSessions: async () => { throw new Error('TDD: not implemented') },
  createSession: async () => { throw new Error('TDD: not implemented') },
  renameSession: async () => { throw new Error('TDD: not implemented') },
  deleteSession: async () => { throw new Error('TDD: not implemented') },
  sendMessage: async () => { throw new Error('TDD: not implemented') },
}))
```

    set((s) => ({

  try {
    await deleteSession(id).send()
    set((s) => ({
      sessions: s.sessions.filter((sess) => sess.id !== id),
      activeSession: s.activeSession?.id === id ? null : s.activeSession,
      isLoadingSessions: false,
    }))
  } catch {
    set({ error: '删除会话失败', isLoadingSessions: false })
  }
},

renameSession: async (id, title) => {
  const trimmed = title.trim()
  if (!trimmed) return

  set({ isLoadingSessions: true, error: null })
  try {
    await renameSession(id, trimmed).send()
    set((s) => ({
      sessions: s.sessions.map((sess) =>
        sess.id === id ? { ...sess, title: trimmed } : sess,
      ),
      isLoadingSessions: false,
    }))
  } catch {
    set({ error: '重命名失败', isLoadingSessions: false })
  }
